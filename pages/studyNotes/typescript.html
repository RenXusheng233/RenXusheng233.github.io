# interface VS type

  • interface - 数据结构
  • type - 类型关系

# 示例

// interface
interface User {
    name: string
    age: number
}

interface SetUser {
    (name: string, age: number): void
}

// type
type User = {
    name: string
    age: number
}

type SetUser = (name: string, age: number) => void

// 拓展

// interface
interface Name {
    name: string
}
interface User extends Name {
    age: number
}
type User = Name & { age: number }
// type
type Name = {
    name: string
}
type User = Name & { age: number }
interface User extends Name {
    age: number
}

// type 可以而 interface 不行

// 基本类型别名, 联合类型, 元组类型
type Name = string

interface Dog {
    wang()
}
interface Cat {
    miao()
}
type Pet = Dog | Cat

type PetList = [Dog, Cat]

// 实例类型赋值
let div = document.createElement('div')
type a = typeof div

// ... 
type StringOrNumber = string | number
type Text = string | { text: string }

// interface 可以而 type 不行
interface User {
    name: string
    age: number
}

interface User {
    sex: string
}

// 实际结果为
// 项目中一般过不了 eslint 检查...
interface User {
    name: string
    age: number
    sex: string
}

# 总结

能用 interface 就用 interface,如果不行就使用 type

# 泛型

T 是一个抽象类型, 只有在调用的时候才确定它的值
泛型就是对类型编程

  • 需要作用到很多类型
  • 需要被用到很多地方

# 示例

// 基本
function id<T>(arg: T): T {
    return arg;
}

function ids<T, U>(arg1: T, arg2: U): [T, U] {
    return [arg1, arg2];
}

// 泛型约束
interface Sizeable {
    size: number
}
function trace<T extends Sizeable>(arg: T): T {
    console.log(arg.size)
    return arg
}

// 默认参数
type A<T = string> = Array<T>

const a: A = [1] // type error
const b: A = ['1'] // ok
const c: A<number> = [1] // ok

# 泛型工具及实现(部分)

// Partial 类型属性变为可选
type Partial<T> = { [P in keyof T ]?: T[P] }

// Required 类型属性变为必填
type Required<T> = { [P in keyof T]-?: T[P] }

// Mutable 类型属性变为可修改
type Mutable<T> = { -readonly [P in keyof T]: T[P] }

// Readonly 类型属性变为可读
type Readonly<T> = { readonly [P in keyof T]: T[P] }

// ReturnType 得到函数的返回值类型
type ReturnType<T extends (...args: any[]) => any> = T extends (
    ...args: any[]
) => infer R
    ? R
    : any

# infer

infer 可以在extends 的条件语句中推断待推断的类型

  • 推断返回值
  • 解包

# 示例

type Ids = number[]
type Names = string []

type Unpacked = T extends Names ? string : T extends Ids ? number : T
// 等价于
type Unpacked = T extends (infer R)[] ? R : T

type IdType = Unpacked<Ids> // number
type NameType = Unpacked<Names> // string

type Response = Promise<number[]>
type Unpacked = T extends Promise<infer R> ? R : T

type ResType = Unpacked<Response> // number[]

type Foo<T> = T extends { a: infer U, b: infer U } ? U : never
type T10 = Foo<{ a: string, b: string }> // string
type T11 = Foo<{ a: string, b: number }> // string | number

type ElementOf<T> = T extends (infer R)[] ? R : never
type TTuple = [string, number]
type Union = ElementOf<TTuple> // string | number
